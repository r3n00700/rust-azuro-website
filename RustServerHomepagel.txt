    function renderServerStatus() {
      Object.entries(serverStatusData).forEach(([key, data]) => {
        const card = document.querySelector(`[data-server="${key}"]`);
        if (!card) return;
        const pill = card.querySelector('.status-pill');
        if (pill) {
          pill.textContent = data.online ? 'Online' : 'Offline';
          pill.classList.toggle('online', data.online);
          pill.classList.toggle('offline', !data.online);
        }
        const players = document.getElementById(`players-${key}`);
        const max = document.getElementById(`max-${key}`);
        const queue = document.getElementById(`queue-${key}`);
        if (players) players.textContent = data.players;
        if (max) max.textContent = data.max;
        if (queue) queue.textContent = data.queue;
      });
    }
